CREATE TABLE PHARMACEUTICAL_COMPANY (
    NAME VARCHAR(50) PRIMARY KEY,
    PHONE VARCHAR(30),
    P_LOCATION VARCHAR(100));

CREATE TABLE DRUGS
(
    BARCODE VARCHAR(10) PRIMARY KEY,
    NAME VARCHAR(30),
    DOSE DECIMAL(10,0),
    TRADE_PRICE FLOAT,
    QUANTITY INT,
    NO_OF_SHEETS INT,
    PRODUCTION_DATE DATE,
    EXIPRY_DATE DATE
    D_ADDRESS VARCHAR(100));

CREATE TABLE DELIVERS
(
  NAME REFERENCES PHARMACEUTICAL_COMPANY(NAME) ON DELETE CASCADE,
  BARCODE REFERENCES DRUGS(BARCODE) ON DELETE CASCADE,
  D_DATE DATE,PRIMARY KEY(NAME,BARCODE));


CREATE TABLE COMPANY 
(
    BID INT PRIMARY KEY,
    NAME VARCHAR(20),
    BLOCATION VARCHAR(100),
    PHONE VARCHAR(11));

CREATE TABLE RECEIVED
(
   BID REFERENCES COMPANY(BID) ON DELETE CASCADE
   BARCODE REFERENCES DRUGS(BARCODE) ON DELETE CASCADE,
   R_DATE DATE,
   B_ADDRESS VARCHAR(100),
   PRIMARY KEY(BID,BARCODE));

CREATE TABLE EMPLOYEE
(
  EID INT PRIMARY KEY,
  NAME VARCHAR(20),
  JOB VARCHAR(20),
  PHONE VARCHAR(11),
  EMAIL VARCHAR(20),
  SALARY INT,
  DOB DATE,
  ADDRESS VARCHAR(80),
  BID REFERENCES COMPANY(BID) ON DELETE CASCADE );

CREATE TABLE CUSTOMER
(
  CID INT PRIMARY KEY,
  CUSTOMER_NAME VARCHAR(20),
  CUSTOMER_PHONE VARCHAR(10),
  BILL_DATE DATE,
  AMOUNT INT,
  BID REFERENCES COMPANY(BID) ON DELETE CASCADE );
